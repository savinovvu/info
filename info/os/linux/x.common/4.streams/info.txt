0й поток - ввод
1й поток - вывод
2й поток - поток ошибок


# перенаправить поток вывода в файл, ошибка не перенаправляется
ls -l > stdout
ls -l 1> stout

# 2й поток - поток ошибок
ls -l asdljk 2> stderr


# одновременно перенаправить потоки ошибок и стандартный вывод, происходит перезапись файла
ls -l asldfj ./ 2> stderr 1> stdout

#  >> дописать в файл
ls -l asldfj ./ 2>> stderr 1>> stdout

# оба потока направить в один файл, & - перенаправление по указателю
ls -l some_error ./ 1> file_name 2>&1

# 1й поток перенаправлятся в grep, 2й поток направляется в 1й поток
ls -l some_error ./  file_neme 2>&1 | grep asdf


# 2й смотрит в grep, 1й в dev/null
ls -l asdf ./ 2>&1 1> /dev/null | grep asd
